// Хариу үүсгэх алгоритм function generateResponse(message) { const lowerMessage = message.toLowerCase(); // Түлхүүр үг таних if (lowerMessage.includes('aoi')) { return getRandomResponse(responses.aoi); } else if (lowerMessage.includes('сайн')) { return getRandomResponse(responses.greetings); } else if (lowerMessage.includes('шог')) { return getRandomResponse(responses.jokes); } else if (lowerMessage.includes('код')) { return getRandomResponse(responses.code); } else { return getRandomResponse(responses.default); } } // Санамсаргүй хариу сонгох function getRandomResponse(responses) { return responses[Math.floor(Math.random() * responses.length)]; }
